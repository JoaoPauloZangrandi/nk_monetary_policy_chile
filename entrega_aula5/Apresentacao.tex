\section{A pergunta e o modelo}

\begin{frame}{A pergunta da polĂ­tica monetĂˇria}
  \begin{itemize}
    \item O Banco Central move \textbf{um} instrumento --- a \textbf{taxa de juros} (TPM no Chile).
    \item Objetivo: manter a \textbf{inflaĂ§ĂŁo} na \textbf{meta} (3\% a.a.) com a menor oscilaĂ§ĂŁo
          possĂ­vel da \textbf{atividade}.
    \item Para estudar isso de forma disciplinada, usamos um \textbf{modelo Novo-Keynesiano mĂ­nimo}:
          3 equaĂ§Ăµes, 3 variĂˇveis, 3 choques.
  \end{itemize}
  \begin{alertblock}{TrĂŞs variĂˇveis (em desvios do equilĂ­brio)}
    $x_t$ = hiato do produto \quad $\pi_t$ = inflaĂ§ĂŁo \quad $i_t$ = juro nominal (TPM)
  \end{alertblock}
\end{frame}

\begin{frame}{O modelo em trĂŞs equaĂ§Ăµes}
  \begin{block}{IS --- demanda}
    \[ x_t = \mathbb{E}_t x_{t+1} - \tfrac{1}{\sigma}\big(i_t - \mathbb{E}_t\pi_{t+1} - r^\ast\big) + \varepsilon^x_t \]
    Demanda cai quando o \emph{juro real} sobe acima do \emph{neutro} $r^\ast$.
  \end{block}
  \begin{block}{NKPC --- oferta (Phillips)}
    \[ \pi_t = \beta\,\mathbb{E}_t\pi_{t+1} + \kappa\,x_t + \varepsilon^\pi_t \]
    InflaĂ§ĂŁo sobe com o aquecimento ($\kappa$) e com a inflaĂ§ĂŁo esperada.
  \end{block}
  \begin{block}{Regra de Taylor --- o Banco Central}
    \[ i_t = \rho_i\,i_{t-1} + (1-\rho_i)\big(r^\ast + \phi_\pi\pi_t + \phi_x x_t\big) + \varepsilon^i_t \]
    Reage Ă  inflaĂ§ĂŁo ($\phi_\pi$) e ao hiato ($\phi_x$), com suavizaĂ§ĂŁo ($\rho_i$).
  \end{block}
\end{frame}
