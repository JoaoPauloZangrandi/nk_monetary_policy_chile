\section{A pergunta e o modelo}

\begin{frame}{A pergunta da política monetária}
  \begin{itemize}
    \item O Banco Central move \textbf{um} instrumento --- a \textbf{taxa de juros} (TPM no Chile).
    \item Objetivo: manter a \textbf{inflação} na \textbf{meta} (3\% a.a.) com a menor oscilação
          possível da \textbf{atividade}.
    \item Para estudar isso de forma disciplinada, usamos um \textbf{modelo Novo-Keynesiano mínimo}:
          3 equações, 3 variáveis, 3 choques.
  \end{itemize}
  \begin{alertblock}{Três variáveis (em desvios do equilíbrio)}
    $x_t$ = hiato do produto \quad $\pi_t$ = inflação \quad $i_t$ = juro nominal (TPM)
  \end{alertblock}
\end{frame}

\begin{frame}{O modelo em três equações}
  \begin{block}{IS --- demanda}
    \[ x_t = \mathbb{E}_t x_{t+1} - \tfrac{1}{\sigma}\big(i_t - \mathbb{E}_t\pi_{t+1} - r^\ast\big) + \varepsilon^x_t \]
    Demanda cai quando o \emph{juro real} sobe acima do \emph{neutro} $r^\ast$.
  \end{block}
  \begin{block}{NKPC --- oferta (Phillips)}
    \[ \pi_t = \beta\,\mathbb{E}_t\pi_{t+1} + \kappa\,x_t + \varepsilon^\pi_t \]
    Inflação sobe com o aquecimento ($\kappa$) e com a inflação esperada.
  \end{block}
  \begin{block}{Regra de Taylor --- o Banco Central}
    \[ i_t = \rho_i\,i_{t-1} + (1-\rho_i)\big(r^\ast + \phi_\pi\pi_t + \phi_x x_t\big) + \varepsilon^i_t \]
    Reage à inflação ($\phi_\pi$) e ao hiato ($\phi_x$), com suavização ($\rho_i$).
  \end{block}
\end{frame}

\section{Solução e determinação}

\begin{frame}{Resolver = expectativas racionais + Blanchard--Kahn}
  \begin{itemize}
    \item Estado estacionário: $x=\pi=0$, $i=r^\ast$ (modelo em desvios).
    \item \textbf{2 variáveis de salto} ($x,\pi$) e \textbf{1 estado} ($i_{t-1}$) $\Rightarrow$
          precisamos de exatamente 2 raízes explosivas.
    \item Dynare confirma: autovalores $0{,}656$ (estável), $1{,}04$, $1{,}379$ (explosivas)
          $\Rightarrow$ \textbf{solução única e estável}.
  \end{itemize}
  \begin{exampleblock}{Princípio de Taylor}
    A determinação exige $\phi_\pi>1$: o BC tem de reagir à inflação \emph{mais que
    proporcionalmente} para ancorar as expectativas.
  \end{exampleblock}
\end{frame}

\section{IRFs, momentos e FEVD}

\begin{frame}{IRFs do baseline --- resposta a cada choque}
  \fig{irf_baseline_all_shocks.png}
\end{frame}

\begin{frame}{IRFs do baseline --- leitura}
  \begin{itemize}
    \item \textbf{Demanda} ($\varepsilon^x>0$): hiato $\uparrow$, inflação $\uparrow$, juro $\uparrow$.
    \item \textbf{Custo} ($\varepsilon^\pi>0$): inflação $\uparrow$, juro $\uparrow$, hiato $\downarrow$
          --- o \emph{trade-off} clássico.
    \item \textbf{Política} ($\varepsilon^i>0$): aperto $\Rightarrow$ hiato $\downarrow$, inflação
          $\downarrow$; resposta gradual pela suavização $\rho_i$.
    \item Convergência rápida (estabilidade confirmada).
  \end{itemize}
\end{frame}

\begin{frame}{Momentos: modelo \emph{versus} dados}
  \figw{moments_model_vs_data.png}

  {\footnotesize O modelo \textbf{subestima} a volatilidade e a \textbf{persistência da inflação}
  (modelo $\approx 0{,}05$ vs dados $\approx 0{,}4$): a NKPC pura não tem inércia.}
\end{frame}

\begin{frame}{FEVD --- de onde vem a variância de cada variável}
  \fig{fevd_calibration_comparison.png}
\end{frame}

\begin{frame}{FEVD --- leitura}
  \begin{itemize}
    \item \textbf{Hiato}: demanda + política ($\sim$45\% + 45\%).
    \item \textbf{Inflação}: dominada por choques de \textbf{custo} ($\sim$80\%).
    \item \textbf{Juro}: pelo seu \textbf{próprio} choque ($\sim$75\%).
    \item A calibração dos choques \emph{atinge} os alvos (``Target'' $\approx$ ``Achieved'').
  \end{itemize}
\end{frame}

  \begin{block}{Receita das páginas 48--49}
    \texttt{varobs pi i x;}\\
    \texttt{estimation(datafile='...', mode\_compute=0, forecast=8);}
